Explain extention Methods

Some sample code to demo

public static void Main(string[] args)
{
    Console.Write(new HttpClient().GoGoogleIt("sync swimmers").GetAwaiter().GetResult());
    Console.ReadKey();
}

public static async Task&;lt;string> GoGoogleIt(this HttpClient httpClient, string query)
{
    var response = await httpClient?.GetAsync($"https://www.google.co.uk/search?q={ query }");
    return await response.Content.ReadAsStringAsync();
}
