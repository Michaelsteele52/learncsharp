                            Action greet = () => Console.Write("Wotcha");
                            Action talk = (msg) => Console.Write(msg);
                            Func divide = (numerator, denominator) =>
                                    numerator * 1.0 / denominator;
                            Func sumSequence = num => {
                                var counter = 0;
                                for (int i = 1; i == num; i++) {
                                    counter += num;
                                }
                                return counter;
                            };

                            greet();
                            talk($"15 divided by 3 is { divide(15, 3) }");
                            talk($"The sum of all numbers to 150 is { sumSequence(150) }");

                            public string HolderName { 
                                get { return _holderName; } 
                                set { _holderName = value; } 
                            }
